from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler, LabelEncoder

# Remplacer les valeurs manquantes par la mdiane
df['TotalCharges'] = df['TotalCharges'].fillna(df['TotalCharges'].median())


# Sparer les caractristiques (X) et la cible (y)
X = df.drop(columns=['Churn'])
y = df['Churn']

# Encoder la variable cible (Yes -> 1, No -> 0)
label_encoder = LabelEncoder()
y = label_encoder.fit_transform(y)

# Identifier les variables catgoriques
categorical_features = X.select_dtypes(include=['object']).columns

# Appliquer l'encodage one-hot pour les variables catgoriques
X_encoded = pd.get_dummies(X, columns=categorical_features, drop_first=True)

# Standardiser les variables numriques
scaler = StandardScaler()
numerical_features = ['tenure', 'MonthlyCharges', 'TotalCharges']
X_encoded[numerical_features] = scaler.fit_transform(X_encoded[numerical_features])

# Diviser les donnes en ensemble d'entranement et de test
X_train, X_test, y_train, y_test = train_test_split(X_encoded, y, test_size=0.2, random_state=42, stratify=y)

# Vrifier la forme des ensembles
X_train.shape, X_test.shape, y_train.shape, y_test.shape

import matplotlib.pyplot as plt

from sklearn.metrics import roc_curve, auc

# Calcul des probabilits pour la courbe ROC
y_probs = model.predict_proba(X_test)[:, 1]  # Probabilit d'appartenance  la classe 1
fpr, tpr, _ = roc_curve(y_test, y_probs)
roc_auc = auc(fpr, tpr)

# Trac de la courbe ROC
plt.figure(figsize=(8, 6))
plt.plot(fpr, tpr, color='blue', lw=2, label=f'ROC curve (AUC = {roc_auc:.2f})')
plt.plot([0, 1], [0, 1], color='gray', linestyle='--')  # Courbe alatoire
plt.xlim([0.0, 1.0])
